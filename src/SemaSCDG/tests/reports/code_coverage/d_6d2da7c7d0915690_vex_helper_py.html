77 fstring = 'Iop_Cmp%s{arg_t[0]}%s' % (fstring_fragment, signedness)

205 def op_generic(self, Operation, op_generator): 

206 def instance(*args): # Note: The args here are all RdTmps 

207 for arg in args: assert isinstance(arg, RdTmp) or isinstance(arg, Const) 

208 arg_types = [self.get_type(arg) for arg in args] 

209 # two operations should never share the same argument instances, copy them here to ensure that 

210 args = [copy.copy(a) for a in args] 

211 op = Operation(op_generator(arg_types), args) 

212 msg = "operation needs to be well typed: " + str(op) 

213 assert op.typecheck(self.irsb.tyenv), msg + "\ntypes: " + str(self.irsb.tyenv) 

214 return self._settmp(op) 

215 

216 return instance

224 def cast_to(self, rdt, tydest, signed=False, high=False): 

225 goalwidth = get_type_size(tydest)

273 def set_bit(self, rdt, idx, bval): 

274 currbit = self.get_bit(rdt, idx) 

275 bvalbit = self.op_extract_lsb(bval) 

276 areequalextrabits = self.op_xor(bval, currbit) 

277 one = self.mkconst(1, self.get_type(areequalextrabits)) 

278 areequal = self.op_and(areequalextrabits, one)

282 def set_bits(self, rdt, idxsandvals): 

283 ty = self.get_type(rdt) 

284 if all([isinstance(idx, Const) for idx, _ in idxsandvals]): 

285 relevantbits = self.mkconst(sum([1 << idx.con.value for idx, _ in idxsandvals]), ty) 

286 else: 

287 relevantbits = self.mkconst(0, ty) 

288 for idx, _ in idxsandvals: 

289 shifted = self.op_shl(self.mkconst(1, ty), idx) 

290 relevantbits = self.op_or(relevantbits, shifted) 

291 setto = self.mkconst(0, ty) 

292 for idx, bval in idxsandvals: 

293 bvalbit = self.op_extract_lsb(bval) 

294 shifted = self.op_shl(bvalbit, idx) 

295 setto = self.op_or(setto, shifted) 

296 shouldflip = self.op_and(self.op_xor(setto, rdt), relevantbits) 

297 return self.op_xor(rdt, shouldflip)
