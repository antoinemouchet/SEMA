6class Callable(object): 

7 """ 

8 Callable is a representation of a function in the binary that can be 

9 interacted with like a native python function. 

10 

11 If you set perform_merge=True (the default), the result will be returned to you, and 

12 you can get the result state with callable.result_state. 

13 

14 Otherwise, you can get the resulting simulation manager at callable.result_path_group. 

15 """ 

16 

17 def __init__(self, project, addr, prototype=None, concrete_only=False, perform_merge=True, base_state=None, toc=None, 

18 cc=None): 

19 """ 

20 :param project: The project to operate on 

21 :param addr: The address of the function to use 

22 

23 The following parameters are optional: 

24 

25 :param prototype: The signature of the calls you would like to make. This really shouldn't be optional. 

26 :param concrete_only: Throw an exception if the execution splits into multiple paths 

27 :param perform_merge: Merge all result states into one at the end (only relevant if concrete_only=False) 

28 :param base_state: The state from which to do these runs 

29 :param toc: The address of the table of contents for ppc64 

30 :param cc: The SimCC to use for a calling convention 

31 """

33 self._project = project

53 def __call__(self, *args): 

54 prototype = SimCC.guess_prototype(args, self._func_ty).with_arch(self._project.arch) 

55 self.perform_call(*args, prototype=prototype) 

56 if self.result_state is not None and prototype.returnty is not None: 

57 loc = self._cc.return_val(prototype.returnty)

63 def perform_call(self, *args, prototype=None):

65 state = self._project.factory.call_state(self._addr, *args, 

66 prototype=prototype, 

67 cc=self._cc, 

68 base_state=self._base_state, 

69 ret_addr=self._deadend_addr, 

70 toc=self._toc)

72 def step_func(pg): 

73 pg2 = pg.prune() 

74 if len(pg2.active) > 1: 

75 raise AngrCallableMultistateError("Execution split on symbolic condition!") 

76 return pg2

78 caller = self._project.factory.simulation_manager(state) 

79 caller.run(step_func=step_func if self._concrete_only else None).unstash(from_stash='deadended') 

80 caller.prune(filter_func=lambda pt: pt.addr == self._deadend_addr)

82 if len(caller.active) == 0:

111

113 raise AngrCallableError("Error in parsing the given C-style argument string.")

124 # string
