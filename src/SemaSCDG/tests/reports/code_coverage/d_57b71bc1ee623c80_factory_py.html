34 def snippet(self, addr, jumpkind=None, **block_opts): 

35 if self.project.is_hooked(addr) and jumpkind != 'Ijk_NoHook': 

36 hook = self.project._sim_procedures[addr] 

37 size = hook.kwargs.get('length', 0) 

38 return HookNode(addr, size, self.project.hooked_by(addr)) 

39 elif self.project.simos.is_syscall_addr(addr): 

40 syscall = self.project.simos.syscall_from_addr(addr) 

41 size = syscall.kwargs.get('length', 0) 

42 return SyscallNode(addr, size, syscall)

111 def call_state(self, addr, *args, **kwargs): 

112 """ 

113 Returns a state object initialized to the start of a given function, as if it were called with given parameters. 

114 

115 :param addr: The address the state should start at instead of the entry point. 

116 :param args: Any additional positional arguments will be used as arguments to the function call. 

117 

118 The following parameters are optional. 

119 

120 :param base_state: Use this SimState as the base for the new state instead of a blank state. 

121 :param cc: Optionally provide a SimCC object to use a specific calling convention. 

122 :param ret_addr: Use this address as the function's return target. 

123 :param stack_base: An optional pointer to use as the top of the stack, circa the function entry point 

124 :param alloc_base: An optional pointer to use as the place to put excess argument data 

125 :param grow_like_stack: When allocating data at alloc_base, whether to allocate at decreasing addresses 

126 :param toc: The address of the table of contents for ppc64 

127 :param initial_prefix: If this is provided, all symbolic registers will hold symbolic values with names 

128 prefixed by this string. 

129 :param fs: A dictionary of file names with associated preset SimFile objects. 

130 :param concrete_fs: bool describing whether the host filesystem should be consulted when opening files. 

131 :param chroot: A path to use as a fake root directory, Behaves similarly to a real chroot. Used only 

132 when concrete_fs is set to True. 

133 :param kwargs: Any additional keyword args will be passed to the SimState constructor. 

134 :return: The state at the beginning of the function. 

135 :rtype: SimState 

136 

137 The idea here is that you can provide almost any kind of python type in `args` and it'll be translated to a 

138 binary format to be placed into simulated memory. Lists (representing arrays) must be entirely elements of the 

139 same type and size, while tuples (representing structs) can be elements of any type and size. 

140 If you'd like there to be a pointer to a given value, wrap the value in a `SimCC.PointerWrapper`. Any value 

141 that can't fit in a register will be automatically put in a 

142 PointerWrapper. 

143 

144 If stack_base is not provided, the current stack pointer will be used, and it will be updated. 

145 If alloc_base is not provided, the current stack pointer will be used, and it will be updated. 

146 You might not like the results if you provide stack_base but not alloc_base. 

147 

148 grow_like_stack controls the behavior of allocating data at alloc_base. When data from args needs to be wrapped 

149 in a pointer, the pointer needs to point somewhere, so that data is dumped into memory at alloc_base. If you 

150 set alloc_base to point to somewhere other than the stack, set grow_like_stack to False so that sequencial 

151 allocations happen at increasing addresses. 

152 """ 

153 return self.project.simos.state_call(addr, *args, **kwargs)

155 def simulation_manager(self, thing: Optional[Union[List[SimState], SimState]]=None, **kwargs) -> 'SimulationManager': 

156 """ 

157 Constructs a new simulation manager. 

158 

159 :param thing: Optional - What to put in the new SimulationManager's active stash (either a SimState or a list of SimStates). 

160 :param kwargs: Any additional keyword arguments will be passed to the SimulationManager constructor 

161 :returns: The new SimulationManager 

162 :rtype: angr.sim_manager.SimulationManager 

163 

164 Many different types can be passed to this method: 

165 

166 * If nothing is passed in, the SimulationManager is seeded with a state initialized for the program 

167 entry point, i.e. :meth:`entry_state()`. 

168 * If a :class:`SimState` is passed in, the SimulationManager is seeded with that state. 

169 * If a list is passed in, the list must contain only SimStates and the whole list will be used to seed the SimulationManager. 

170 """ 

171 if thing is None: 

172 thing = [ self.entry_state() ] 

173 elif isinstance(thing, (list, tuple)): 

174 if any(not isinstance(val, SimState) for val in thing): 

175 raise AngrError("Bad type to initialize SimulationManager") 

176 elif isinstance(thing, SimState): 

177 thing = [ thing ] 

178 else: 

179 raise AngrError("BadType to initialze SimulationManager: %s" % repr(thing)) 

180 

181 return SimulationManager(self.project, active_states=thing, **kwargs)
