93 def parse_lsda(self, address, offset): 

94 self.address = address 

95 self.base_offset = offset

97 header = self._parse_lsda_header()

99 csrs = [ ] # type: List[CallSiteEntry] 

100 start_offset = self.stream.tell() 

101 while self.stream.tell() - start_offset < header.call_site_table_len: 

102 csr = self._parse_call_site_entry(header.call_site_encoding) 

103 if csr is not None:

108 def _parse_lsda_header(self): 

109 

110 # lpstart 

111 lpstart_encoding = self.stream.read(1)[0] 

112 if lpstart_encoding != DW_EH_encoding_flags['DW_EH_PE_omit']: 

113 base_encoding = lpstart_encoding & 0x0f 

114 modifier = lpstart_encoding & 0xf0 

115 

116 lpstart = struct_parse( 

117 Struct('dummy', 

118 self._formats[base_encoding]('LPStart')), 

119 self.stream 

120 )['LPStart'] 

121 

122 if modifier == 0: 

123 pass 

124 elif modifier == DW_EH_encoding_flags['DW_EH_PE_pcrel']: 

125 lpstart += self.address + (self.stream.tell() - self.base_offset) 

126 else: 

127 raise NotImplementedError("Unsupported modifier %#x." % modifier) 

128 

129 else: 

130 lpstart = None 

131 

132 # ttype 

133 ttype_encoding = self.stream.read(1)[0] 

134 if ttype_encoding != DW_EH_encoding_flags['DW_EH_PE_omit']: 

135 ttype_offset = struct_parse( 

136 Struct('dummy', 

137 self.entry_structs.Dwarf_uleb128('TType')), 

138 self.stream 

139 )['TType'] 

140 else: 

141 ttype_offset = None

143 # call site table length 

144 cstable_encoding = self.stream.read(1)[0]
