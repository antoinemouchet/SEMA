279 def read(self, fmt: Union[int, str]) -> Union[int, float, str, Bits, bool, bytes, None]: 

280 """Interpret next bits according to the format string and return result. 

281 

282 fmt -- Token string describing how to interpret the next bits. 

283 

284 Token examples: 'int:12' : 12 bits as a signed integer 

285 'uint:8' : 8 bits as an unsigned integer 

286 'float:64' : 8 bytes as a big-endian float 

287 'intbe:16' : 2 bytes as a big-endian signed integer 

288 'uintbe:16' : 2 bytes as a big-endian unsigned integer 

289 'intle:32' : 4 bytes as a little-endian signed integer 

290 'uintle:32' : 4 bytes as a little-endian unsigned integer 

291 'floatle:64': 8 bytes as a little-endian float 

292 'intne:24' : 3 bytes as a native-endian signed integer 

293 'uintne:24' : 3 bytes as a native-endian unsigned integer 

294 'floatne:32': 4 bytes as a native-endian float 

295 'hex:80' : 80 bits as a hex string 

296 'oct:9' : 9 bits as an octal string 

297 'bin:1' : single bit binary string 

298 'ue' : next bits as unsigned exp-Golomb code 

299 'se' : next bits as signed exp-Golomb code 

300 'uie' : next bits as unsigned interleaved exp-Golomb code 

301 'sie' : next bits as signed interleaved exp-Golomb code 

302 'bits:5' : 5 bits as a bitstring 

303 'bytes:10' : 10 bytes as a bytes object 

304 'bool' : 1 bit as a bool 

305 'pad:3' : 3 bits of padding to ignore - returns None 

306 

307 fmt may also be an integer, which will be treated like the 'bits' token. 

308 

309 The position in the bitstring is advanced to after the read items. 

310 

311 Raises ReadError if not enough bits are available. 

312 Raises ValueError if the format is not understood. 

313 

314 """ 

315 if isinstance(fmt, numbers.Integral): 

316 if fmt < 0: 

317 raise ValueError("Cannot read negative amount.") 

318 if fmt > self.len - self._pos: 

319 raise ReadError(f"Cannot read {fmt} bits, only {self.len - self._pos} available.") 

320 bs = self._slice(self._pos, self._pos + fmt) 

321 self._pos += fmt 

322 return bs 

323 p = self._pos 

324 _, token = tokenparser(fmt) 

325 if len(token) != 1: 

326 self._pos = p 

327 raise ValueError(f"Format string should be a single token, not {len(token)} " 

328 "tokens - use readlist() instead.") 

329 name, length, _ = token[0] 

330 try: 

331 value, self._pos = self._readtoken(name, self._pos, length) 

332 except ValueError as e: 

333 raise InterpretError(e) 

334 return value

338 """Interpret next bits according to format string(s) and return list. 

339 

340 fmt -- A single string or list of strings with comma separated tokens 

341 describing how to interpret the next bits in the bitstring. Items 

342 can also be integers, for reading new bitstring of the given length.

361 def readto(self: TConstBitStream, bs: BitsType, bytealigned: Optional[bool] = None) -> TConstBitStream: 

362 """Read up to and including next occurrence of bs and return result. 

363 

364 bs -- The bitstring to find. An integer is not permitted. 

365 bytealigned -- If True the bitstring will only be 

366 found on byte boundaries. 

367 

368 Raises ValueError if bs is empty. 

369 Raises ReadError if bs is not found. 

370 

371 """ 

372 if isinstance(bs, numbers.Integral): 

373 raise ValueError("Integers cannot be searched for") 

374 bs = Bits._create_from_bitstype(bs) 

375 oldpos = self._pos 

376 p = self.find(bs, self._pos, bytealigned=bytealigned) 

377 if not p: 

378 raise ReadError("Substring not found") 

379 self._pos += bs.len 

380 return self._slice(oldpos, self._pos)

630 def insert(self, bs: BitsType, pos: Optional[int] = None) -> None: 

631 """Insert bs at bit position pos. 

632 

633 bs -- The bitstring to insert. 

634 pos -- The bit position to insert at. 

635 

636 The current bit position will be moved to the end of the inserted section. 

637 Raises ValueError if pos < 0 or pos > len(self). 

638 

639 """ 

640 bs = Bits._create_from_bitstype(bs) 

641 if len(bs) == 0: 

642 return 

643 if bs is self: 

644 bs = self._copy() 

645 if pos is None: 

646 pos = self._pos 

647 if pos < 0: 

648 pos += self._getlength() 

649 if not 0 <= pos <= self._getlength(): 

650 raise ValueError("Invalid insert position.") 

651 self._insert(bs, pos) 

652 self._pos = pos + len(bs)

654 def replace(self, old: BitsType, new: BitsType, start: Optional[int] = None, end: Optional[int] = None, 

655 count: Optional[int] = None, bytealigned: Optional[bool] = None) -> int: 

656 """Replace all occurrences of old with new in place. 

657 

658 Returns number of replacements made. 

659 

660 old -- The bitstring to replace. 

661 new -- The replacement bitstring. 

662 start -- Any occurrences that start before this will not be replaced. 

663 Defaults to 0. 

664 end -- Any occurrences that finish after this will not be replaced. 

665 Defaults to len(self). 

666 count -- The maximum number of replacements to make. Defaults to 

667 replace all occurrences. 

668 bytealigned -- If True replacements will only be made on byte 

669 boundaries. 

670 

671 Raises ValueError if old is empty or if start or end are 

672 out of range. 

673 

674 """ 

675 if count == 0: 

676 return 0 

677 old = Bits._create_from_bitstype(old) 

678 new = Bits._create_from_bitstype(new) 

679 if not old.len: 

680 raise ValueError("Empty bitstring cannot be replaced.") 

681 start, end = self._validate_slice(start, end) 

682 if new is self: 

683 # Prevent self assignment woes 

684 new = copy.copy(self) 

685 length_before = len(self) 

686 replacement_count = self._replace(old, new, start, end, 0 if count is None else count, bytealigned) 

687 if len(self) != length_before: 

688 self._pos = 0 

689 return replacement_count
