18def lift(data, addr, arch, max_bytes=None, max_inst=None, bytes_offset=0, opt_level=1, traceflags=0, 

19 strict_block_end=True, inner=False, skip_stmts=False, collect_data_refs=False, cross_insn_opt=True, 

20 load_from_ro_regions=False): 

21 """ 

22 Recursively lifts blocks using the registered lifters and postprocessors. Tries each lifter in the order in 

23 which they are registered on the data to lift. 

24 

25 If a lifter raises a LiftingException on the data, it is skipped. 

26 If it succeeds and returns a block with a jumpkind of Ijk_NoDecode, all of the lifters are tried on the rest 

27 of the data and if they work, their output is appended to the first block. 

28 

29 :param arch: The arch to lift the data as. 

30 :type arch: :class:`archinfo.Arch` 

31 :param addr: The starting address of the block. Effects the IMarks. 

32 :param data: The bytes to lift as either a python string of bytes or a cffi buffer object. 

33 :param max_bytes: The maximum number of bytes to lift. If set to None, no byte limit is used. 

34 :param max_inst: The maximum number of instructions to lift. If set to None, no instruction limit is used. 

35 :param bytes_offset: The offset into `data` to start lifting at. 

36 :param opt_level: The level of optimization to apply to the IR, -1 through 2. -1 is the strictest 

37 unoptimized level, 0 is unoptimized but will perform some lookahead/lookbehind 

38 optimizations, 1 performs constant propogation, and 2 performs loop unrolling, 

39 which honestly doesn't make much sense in the context of pyvex. The default is 1. 

40 :param traceflags: The libVEX traceflags, controlling VEX debug prints. 

41 

42 .. note:: Explicitly specifying the number of instructions to lift (`max_inst`) may not always work 

43 exactly as expected. For example, on MIPS, it is meaningless to lift a branch or jump 

44 instruction without its delay slot. VEX attempts to Do The Right Thing by possibly decoding 

45 fewer instructions than requested. Specifically, this means that lifting a branch or jump 

46 on MIPS as a single instruction (`max_inst=1`) will result in an empty IRSB, and subsequent 

47 attempts to run this block will raise `SimIRSBError('Empty IRSB passed to SimIRSB.')`. 

48 

49 .. note:: If no instruction and byte limit is used, pyvex will continue lifting the block until the block 

50 ends properly or until it runs out of data to lift. 

51 """ 

52 if max_bytes is not None and max_bytes <= 0: 

53 raise PyVEXError("Cannot lift block with no data (max_bytes <= 0)") 

54 

55 if not data: 

56 raise PyVEXError("Cannot lift block with no data (data is empty)")

58 if isinstance(data, str):

64 allow_arch_optimizations = False

67 raise PyVEXError("Cannot lift block with ffi pointer and no size (max_bytes is None)") 

68 c_data = data 

69 py_data = None 

70 allow_arch_optimizations = True 

71 

72 # In order to attempt to preserve the property that 

73 # VEX lifts the same bytes to the same IR at all times when optimizations are disabled 

74 # we hack off all of VEX's non-IROpt optimizations when opt_level == -1. 

75 # This is intended to enable comparisons of the lifted IR between code that happens to be 

76 # found in different contexts. 

77 if opt_level < 0: 

78 allow_arch_optimizations = False 

79 opt_level = 0 

80 

81 for lifter in lifters[arch.name]: 

82 try: 

83 u_data = data 

84 if lifter.REQUIRE_DATA_C: 

85 if c_data is None: 

86 u_data = ffi.from_buffer(ffi.BVoidP, py_data + b'\0' * 8 if type(py_data) is bytes else py_data) 

87 max_bytes = min(len(py_data), max_bytes) if max_bytes is not None else len(py_data) 

88 else: 

89 u_data = c_data 

90 skip = 0 

91 elif lifter.REQUIRE_DATA_PY: 

92 if bytes_offset and archinfo.arch_arm.is_arm_arch(arch) and (addr & 1) == 1: 

93 skip = bytes_offset - 1 

94 else: 

95 skip = bytes_offset 

96 if py_data is None: 

97 if max_bytes is None: 

98 l.debug('Cannot create py_data from c_data when no max length is given') 

99 continue 

100 u_data = ffi.buffer(c_data + skip, max_bytes)[:] 

101 else: 

102 if max_bytes is None: 

103 u_data = py_data[skip:] 

104 else: 

105 u_data = py_data[skip:skip + max_bytes] 

106 else: 

107 raise RuntimeError("Incorrect lifter configuration. What type of data does %s expect?" 

108 % lifter.__class__) 

109 

110 try: 

111 final_irsb = lifter(arch, addr)._lift(u_data, bytes_offset - skip, max_bytes, max_inst, opt_level, 

112 traceflags, allow_arch_optimizations, strict_block_end, 

113 skip_stmts, 

114 collect_data_refs=collect_data_refs, 

115 cross_insn_opt=cross_insn_opt, 

116 load_from_ro_regions=load_from_ro_regions, 

117 ) 

118 except SkipStatementsError: 

119 assert skip_stmts is True 

120 final_irsb = lifter(arch, addr)._lift(u_data, bytes_offset - skip, max_bytes, max_inst, opt_level, 

121 traceflags, allow_arch_optimizations, strict_block_end, 

122 skip_stmts=False, 

123 collect_data_refs=collect_data_refs, 

124 cross_insn_opt=cross_insn_opt, 

125 load_from_ro_regions=load_from_ro_regions, 

126 ) 

127 #l.debug('block lifted by %s' % str(lifter)) 

128 #l.debug(str(final_irsb)) 

129 break 

130 except LiftingException as ex: 

131 l.debug('Lifting Exception: %s', str(ex)) 

132 continue 

133 else: 

134 final_irsb = IRSB.empty_block(arch, 

135 addr, 

136 size=0, 

137 nxt=Const(const.vex_int_class(arch.bits)(addr)), 

138 jumpkind='Ijk_NoDecode', 

139 ) 

140 final_irsb.invalidate_direct_next() 

141 return final_irsb 

142 

143 if final_irsb.size > 0 and final_irsb.jumpkind == 'Ijk_NoDecode': 

144 # We have decoded a few bytes before we hit an undecodeable instruction. 

145 

146 # Determine if this is an intentional NoDecode, like the ud2 instruction on AMD64 

147 nodecode_addr_expr = final_irsb.next 

148 if type(nodecode_addr_expr) is Const: 

149 nodecode_addr = nodecode_addr_expr.con.value 

150 next_irsb_start_addr = addr + final_irsb.size 

151 if nodecode_addr != next_irsb_start_addr: 

152 # The last instruction of the IRSB has a non-zero length. This is an intentional NoDecode. 

153 # The very last instruction has been decoded 

154 final_irsb.jumpkind = 'Ijk_NoDecode' 

155 final_irsb.next = final_irsb.next 

156 final_irsb.invalidate_direct_next() 

157 return final_irsb 

158 

159 # Decode more bytes 

160 if skip_stmts: 

161 # When gymrat will be invoked, we will merge future basic blocks to the current basic block. In this case, 

162 # statements are usually required. 

163 # TODO: In the future, we may further optimize it to handle cases where getting statements in gymrat is not 

164 # TODO: required. 

165 return lift(data, addr, arch, 

166 max_bytes=max_bytes, 

167 max_inst=max_inst, 

168 bytes_offset=bytes_offset, 

169 opt_level=opt_level, 

170 traceflags=traceflags, 

171 strict_block_end=strict_block_end, 

172 skip_stmts=False, 

173 collect_data_refs=collect_data_refs, 

174 ) 

175 

176 next_addr = addr + final_irsb.size 

177 if max_bytes is not None: 

178 max_bytes -= final_irsb.size 

179 if isinstance(data, (bytes, bytearray, memoryview)): 

180 data_left = data[final_irsb.size:] 

181 else: 

182 data_left = data + final_irsb.size 

183 if max_inst is not None: 

184 max_inst -= final_irsb.instructions 

185 if (max_bytes is None or max_bytes > 0) and (max_inst is None or max_inst > 0) and data_left: 

186 more_irsb = lift(data_left, next_addr, arch, 

187 max_bytes=max_bytes, 

188 max_inst=max_inst, 

189 bytes_offset=bytes_offset, 

190 opt_level=opt_level, 

191 traceflags=traceflags, 

192 strict_block_end=strict_block_end, 

193 inner=True, 

194 skip_stmts=False, 

195 collect_data_refs=collect_data_refs, 

196 ) 

197 if more_irsb.size: 

198 # Successfully decoded more bytes 

199 final_irsb.extend(more_irsb) 

200 elif max_bytes == 0: 

201 # We have no more bytes left. Mark the jumpkind of the IRSB as Ijk_Boring 

202 if final_irsb.size > 0 and final_irsb.jumpkind == 'Ijk_NoDecode': 

203 final_irsb.jumpkind = 'Ijk_Boring' 

204 final_irsb.next = Const(vex_int_class(arch.bits)(final_irsb.addr + final_irsb.size)) 

205 

206 if not inner: 

207 for postprocessor in postprocessors[arch.name]: 

208 try: 

209 postprocessor(final_irsb).postprocess() 

210 except NeedStatementsNotification: 

211 # The post-processor cannot work without statements. Re-lift the current block with skip_stmts=False 

212 if not skip_stmts: 

213 # sanity check 

214 # Why does the post-processor raise NeedStatementsNotification when skip_stmts is False? 

215 raise TypeError("Bad post-processor %s: " 

216 "NeedStatementsNotification is raised when statements are available." % 

217 postprocessor.__class__) 

218 

219 # Re-lift the current IRSB 

220 return lift(data, addr, arch, 

221 max_bytes=max_bytes, 

222 max_inst=max_inst, 

223 bytes_offset=bytes_offset, 

224 opt_level=opt_level, 

225 traceflags=traceflags, 

226 strict_block_end=strict_block_end, 

227 inner=inner, 

228 skip_stmts=False, 

229 collect_data_refs=collect_data_refs, 

230 ) 

231 except LiftingException: 

232 continue 

233 

234 return final_irsb

237def register(lifter, arch_name): 

238 """ 

239 Registers a Lifter or Postprocessor to be used by pyvex. Lifters are are given priority based on the order 

240 in which they are registered. Postprocessors will be run in registration order. 

241 

242 :param lifter: The Lifter or Postprocessor to register 

243 :vartype lifter: :class:`Lifter` or :class:`Postprocessor` 

244 """ 

245 if issubclass(lifter, Lifter): 

246 l.debug("Registering lifter %s for architecture %s.", lifter.__name__, arch_name) 

247 lifters[arch_name].append(lifter) 

248 if issubclass(lifter, Postprocessor): 

249 l.debug("Registering postprocessor %s for architecture %s.", lifter.__name__, arch_name) 

250 postprocessors[arch_name].append(lifter)
